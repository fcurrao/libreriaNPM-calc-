Description: 
Es una calculadora que cuenta con 4 operaciones basicas

Suma

resta

multiplicaicon

division

----------

export const suma = (num1, num2) => {
    return num1 + num2;
  };
  
  export const resta = (num1, num2) => {
    return num1 - num2;
  };
  
  export const mult = (num1, num2) => {
    return num1 * num2;
  };
  
  export const div = (num1, num2) => {
    return num1 / num2;
  };
